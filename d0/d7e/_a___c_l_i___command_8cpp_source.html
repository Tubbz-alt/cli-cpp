111 /* Check Name Substring Match */
112 /*******************************************/
113 bool A_CLI_Command::Is_Name_Substring( const std::string& input_string,
114  std::string& match_result )const
115 {
116  // Match list
117  std::vector<std::string> match_list;
118 
119  // Iterate over the name list
120  for( size_t i=0; i<m_names.size(); i++ )
121  {
122 
123  // If the name is less than the input, it cannot match
124  if( m_names[i].size() < input_string.size() ){
125  continue;
126  }
127 
128  // If the sizes are the same, do a string compare
129  else if( m_names[i].size() == input_string.size() ){
130  if( m_names[i] == input_string ){
131  match_list.push_back(m_names[i]);
132  }
133  else{
134  continue;
135  }
136  }
137 
138  // Otherwise, the name is greater than the input, so check the substring match
139  else{
140 
141  // Look for a perfect substring match
142  if( m_names[i].substr(0, input_string.size()) == input_string ){
143  match_list.push_back(m_names[i]);
144  }
145  }
146  }
147 
148  // Check if we found any matches
149  if( (int)match_list.size() <= 0 ){
150  match_result = "";
151  return false;
152  }
153 
154  // Find the minimum spanning substring
155  match_result = match_list[0];
156  for( size_t i=1; i<match_list.size(); i++ ){
157  match_result = UTILS::String_Substring_Match( match_result, match_list[i]);
158  }
159 
160  // success
161  return true;
162 }
