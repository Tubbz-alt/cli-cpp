60

62  std::string formatted_input = UTILS::Format_Command_String( test_input );
63 
64  // If we are matching the whole, just do a compare
65  if( match_whole == true ){

67  formatted_output = m_command_text;
68  return true;
69  } else {
70  return false;
71  }
72  }
73

76  if( m_alias_name.size() > formatted_input.size() ){
77  return false;
78  }
79 
80  // If the alias is the same size, do a compare
81  if( m_alias_name.size() == formatted_input.size() ){
82  if( m_alias_name == formatted_input ){
83  formatted_output = m_command_text;
84  return true;
85  }
86  else{
87  return false;
88  }
89  }
90 
91  // Lastly, do a substring if smaller
92  if( formatted_input.substr(0, m_alias_name.size()) == m_alias_name ){
93

95  formatted_output = m_command_text + formatted_input.substr(m_alias_name.size());
96  return true;
97  }
