365 {

368 
369  // Set the buffer row
370  int cli_row = m_render_driver_context->Get_Window_Rows() - 2;
371 
372  // Get the cursor text
373  const std::string cursor_text = m_render_state->Get_Cursor_Text();
374  const int cli_prompt_pos = m_render_state->Get_Cursor_Pos();
375 
376  // Write the Cursor Text
377  std::string output = " ";
378  output += UTILS::ANSI_GREEN + std::string("cmd: ") + UTILS::ANSI_RESET;
379 
380  // Don't do this if the cursor is at 0
381  if( cli_prompt_pos > 0 ){
382  output += cursor_text.substr(0,cli_prompt_pos);
383  }
384 
385  // Print the text
386  if( (int)cursor_text.size() > 0 && cli_prompt_pos < (int)cursor_text.size() ){
387  output += UTILS::ANSI_BACK_WHITE + UTILS::ANSI_BLACK + cursor_text[cli_prompt_pos] + UTILS::ANSI_RESET;
388  } else {
389  output += UTILS::ANSI_BACK_WHITE + UTILS::ANSI_BLACK + " " + UTILS::ANSI_RESET;
390  }
391 
392  // Don't do this if we are at the end
393  if( cli_prompt_pos < (int)cursor_text.size()){
394  output += cursor_text.substr(cli_prompt_pos+1);
395  }
396 
397 
398  // Copy to the buffer
399  print_buffer[cli_row+0] = output + BUFFER_NEWLINE + UTILS::ANSI_CURSORINVIS;
400 
401  // Log Exit
402  BOOST_LOG_TRIVIAL(trace) << "End of " << __func__ << " method. File: " << __FILE__ << ", Line: " << __LINE__;
