159 bool A_Command_Argument::Is_Argument_Substring( const std::string& argument_name,
160  std::string& match_name )const
161 {
162  // List of potential matches
163  std::vector<std::string> match_list;
164 
165  // Iterate over autocomplete terms
166  for( size_t i=0; i<m_autocomplete_terms.size(); i++ )
167  {
168  // Check the size
169  if( argument_name.size() > m_autocomplete_terms[i].size() ){
170  continue;
171  }
172 
173  // Check if equal
174  if( argument_name == m_autocomplete_terms[i] ){
175  match_list.push_back(m_autocomplete_terms[i]);
176  continue;
177  }
178 
179  // Check substring
180  if( m_autocomplete_terms[i].substr(0, argument_name.size()) == argument_name )
181  {
182  match_list.push_back(m_autocomplete_terms[i]);
183  continue;
184  }
185  }
186 
187  // If the type if a filesystem type, get a list of files
188  if( m_type == CommandArgumentType::PATH && m_autocomplete_path == true )
189  {
190 
191  // Get a list of paths
192  std::vector<std::string> pathname_list = FS::Get_Contents( argument_name );
193 
194  // iterate over the list, checking the names
195  for( size_t i=0; i<pathname_list.size(); i++ ){
196 
197  // If the pathname is shorter, it cannot be valid
198  if( pathname_list[i].size() < argument_name.size() ){
199  continue;
200  }
201 
202  // Otherwise, do a substring check
203  std::string temp_path = pathname_list[i].substr(0, argument_name.size());
204  if( temp_path == argument_name ){
205  match_list.push_back( pathname_list[i] );
206  }
207  }
208  }
209 
210  // Check if we found any matches
211  if( match_list.size() <= 0 ){
212  match_name = "";
213  return false;
214  }
215 
216  // Find the minimum spanning substring
217  match_name = match_list[0];
218  for( size_t i=1; i<match_list.size(); i++ ){
219  match_name = UTILS::String_Substring_Match( match_name, match_list[i]);
220  }
221  return true;
222 }
