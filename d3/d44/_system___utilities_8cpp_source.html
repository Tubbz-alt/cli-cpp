27 {
28  // Build stream
29  std::stringstream sin;
30  int max_frames = 65;
31 
32  // Print
33  sin << "stack trace:\n";
34 
35  // storage array for stack trace address data
36  void* addrlist[max_frames+1];
37 
38  // retrieve current stack addresses
39  int addrlen = backtrace(addrlist, sizeof(addrlist) / sizeof(void*));
40 
41  if (addrlen == 0) {
42  sin << " <empty, possibly corrupt>\n";
43  return;
44  }
45 
46  // resolve addresses into strings containing "filename(function+address)",
47  // this array must be free()-ed
48  char** symbollist = backtrace_symbols(addrlist, addrlen);
49 
50  // allocate string which will be filled with the demangled function name
51  size_t funcnamesize = 256;
52  char* funcname = (char*)malloc(funcnamesize);
53 
54  // iterate over the returned symbol lines. skip the first, it is the
55  // address of this function.
56  for (int i = 1; i < addrlen; i++)
57  {
58  char *begin_name = 0, *begin_offset = 0, *end_offset = 0;
59 
60  // find parentheses and +address offset surrounding the mangled name:
61  // ./module(function+0x15c) [0x8048a6d]
62  for (char *p = symbollist[i]; *p; ++p)
63  {
64  if (*p == '(')
65  begin_name = p;
66  else if (*p == '+')
67  begin_offset = p;
68  else if (*p == ')' && begin_offset) {
69  end_offset = p;
70  break;
71  }
72  }
73 
74  if (begin_name && begin_offset && end_offset
75  && begin_name < begin_offset)
76  {
77  *begin_name++ = '\0';
78  *begin_offset++ = '\0';
79  *end_offset = '\0';
80 
81  // mangled name is now in [begin_name, begin_offset) and caller
82  // offset in [begin_offset, end_offset). now apply
83  // __cxa_demangle():
84 
85  int status;
86  char* ret = abi::__cxa_demangle(begin_name,
87  funcname, &funcnamesize, &status);
88  if (status == 0) {
89  funcname = ret; // use possibly realloc()-ed string
90  sin << symbollist[i] << " : " << funcname << "+" << begin_offset << std::endl;
91  }
92  else {
93  // demangling failed. Output function name as a C function with
94  // no arguments.
95  sin << " " << symbollist[i] << " : " << begin_name << "() + " << begin_offset << std::endl;
96  }
97  }
98  else
99  {
100  // couldn't parse the line? print the whole line.
101  sin << " " << symbollist[i] << std::endl;
102  }
103  }
104 
105  free(funcname);
106  free(symbollist);
107 
108  // Log Output
109  BOOST_LOG_TRIVIAL(fatal) << sin.str();
110 
111 }
